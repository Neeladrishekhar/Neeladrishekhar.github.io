    for (int i=0; i<numberOfRows; i++){
        vector<int> a;
        for (int j=0; j<numberOfColumns; j++){
        int x;


        cin>>x;
        a.push_back(x);
        }
        matrix.push_back(a);
}

    }
    
    return 0;
}
