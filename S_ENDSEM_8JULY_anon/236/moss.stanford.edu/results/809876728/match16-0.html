#include <iostream>
using namespace std;

struct Point {
    float x;
    float y;
};

// write code for the function findCentroid()


void findCentroid(const Point One,const Point Two,const Point Three,float &k,float &l){
    k=(One.x+Two.x+Three.x)/3;
    l=(One.y+Two.y+Three.y)/3;
    cout<<k<<" "<<l<<endl;
}



int main () {
    // read in 4 points and call findCentroid and print centroid coordinates
    Point One,Two,Three;
    cin>>One.x>>One.y;
    cin>>Two.x>>Two.y;
    cin>>Three.x>>Three.y;
    float k,l;
    
    findCentroid(One,Two,Three,k,l);
    
    // cout<<k<<" "<<l<<endl;
    
    return 0;
}
