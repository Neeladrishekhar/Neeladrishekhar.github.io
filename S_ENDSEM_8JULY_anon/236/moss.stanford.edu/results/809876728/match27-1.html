#include <iostream>
using namespace std;

struct Point {
    float x;
    float y;
};
void findCentroid(Point one,Point two,Point three,Point Centroid){
    
    Centroid.x=(one.x+two.x+three.x)/3;
    Centroid.y=(one.y+two.y+three.y)/3;


    cout<<Centroid.x<<" "<<Centroid.y;
}
// write code for the function findCentroid()

int main () {
    // read in 4 points and call findCentroid and print centroid coordinates
    Point one,two,three,Centroid;
    cin>>one.x>>one.y;
    cin>>two.x>>two.y;
    cin>>three.x>>three.y;
    findCentroid(one,two,three,Centroid);
    return 0;
}
