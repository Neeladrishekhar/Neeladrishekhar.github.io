#include <iostream>


using namespace std;

struct Point {
    float x;
    float y;
};

// write code for the function findCentroid()
void findCentroid( Point one , Point two , Point three , Point &four) {
    
    four.x = one.x + two.x + three.x ;
    four.y = one.y + two.y + three.y ;

    
}

int main () {
    // read in 4 points and call findCentroid and print centroid coordinates
    Point one , two , three , four ;
    cin >> one.x >> one.y ;
    cin >> two.x >> two.y ;
    cin >> three.x >> three.y ;
    findCentroid( one , two , three , four) ;
    cout << four.x/3.0 << " " << four.y/3.0 ;
    return 0;
}
