#include <iostream>
using namespace std;

struct Point {
    float x;
    float y;
};
void centroid(Point one, Point two, Point three, float &x, float &y){
    x = (one.x + two.x + three.x)/3;
    y = (one.y + two.y + three.y)/3;
}

int main () {
    float x, y;
    Point one, two, three;
    cin>>one.x>>one.y;


    cin>>two.x>>two.y;
    cin>>three.x>>three.y;
    centroid(one, two, three, x, y);
    cout<<x<<" "<<y;
    return 0;
}
