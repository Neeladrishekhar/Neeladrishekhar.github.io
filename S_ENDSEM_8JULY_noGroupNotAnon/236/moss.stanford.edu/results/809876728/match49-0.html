#include <iostream>
using namespace std;

struct Point {
    float x;
    float y;
};

// write code for the function findCentroid()
 Point a,b,c,centroid;
 float findCentroid(a,b,c,centroid){


    centroid.x=(a.x+b.x+c.x)/3;
    centroid.y=(a.y+b.y+c.y)/3;
    
}

int main () {
    // read in 4 points and call findCentroid and print centroid coordinate
    Point a,b,c,centroid;
    cin>>a.x>>" ">>a.y;
    cin>>b.x>>" ">>b.y;
    cin>>c.x>>" ">>c.y;
    findCentroid(a,b,c,centroid);
    cout<<centroid.x<<centroid.y<<endl;
    return 0;
}
