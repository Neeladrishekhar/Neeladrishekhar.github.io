#include <iostream>
using namespace std;

struct Point {
    float x;
    float y;
};

// write code for the function findCentroid()
void findCentroid(Point one,Point two,Point three,Point& four4 ){
    four4.x=(one.x+two.x+three.x)/3;
    four4.y=(one.y+two.y+three.y)/3;
}

int main () {
    // read in 4 points and call findCentroid and print centroid coordinates
    Point one,two,three,four;
    cin>>one.x>>one.y;
    cin>>two.x>>two.y;


    cin>>three.x>>three.y;
    cin>>four.x>>four.y;
    findCentroid(one,two,three,four);
    cout<<four.x<<" "<<four.y<<endl;
    return 0;
}
