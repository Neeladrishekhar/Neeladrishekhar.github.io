#include <iostream>
using namespace std;

struct Point {
    float x;
    float y;
};

void findCentroid ( float Point one, float Point two, float Point three,  Point &four )
{


four.x=( one.x + two.x + three.x)/3;
four.y= (one.y + two.y + three.y)/3;
    
}


// write code for the function findCentroid()

int main () {
    Point one, two, three, four;
    cin>>one.x>>one.y;
    cin>>two.x>>two.y;
    cin>>three.x>>three.y;
    findCentroid(one, two, three, four);
    cout<<four.x<<" "<<four.y<<end;
    
    // read in 4 points and call findCentroid and print centroid coordinates
    return 0;
}
