struct Point {
    float x;
    float y;
};

void findCentroid(float x1 , float y1 ,float x2 , float y2 ,float x3 , float y3  ){
    cout<<(x1+x2+x3)/3<<" "<<(y1+y2+y3)/3;
}

int main () {
    // read in 4 points and call findCentroid and print centroid coordinates
    Point p1,p2 ,p3;
