#include <iostream>
using namespace std;

struct Point {
    float x;
    float y;
};



void findCentroid(Point m,Point n,Point o,Point centroid)
{
    centroid.x=(m.x+n.x+o.x)/3;
    centroid.y=(m.y+n.y+o.y)/3;
    return ;
}
int main () {
    Point m,n,o,centroid;
    cin>>m.x>>m.y;
    cin>>n.x>>n.y;
    cin>>o.x>>o.y;

    return 0;
}
